#include "string.h"


char * invert(char *start,char *end)
{

	char tmp,*ptmp = start;
	while (start != NULL &&end != NULL && start < end)
	{
		tmp = (*start);
		(*start) = (*end);
		(*end) = tmp;
		start++;
		end--;

	}
	return ptmp;
}

char *left(char *s, int pos)
{
	int len = strlen(s);
	invert(s,s + (pos - 1));
	invert(s + pos, s + (len -1));
	invert(s, s + (len - 1));
	return s;
}
