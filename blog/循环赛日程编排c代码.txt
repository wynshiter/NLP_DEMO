


#include <stdio.h>
	
int m,s,n;
int i,j;
int a[100][100];

int sf(int n)
{
	if(n%2!=0)
		printf("输入参数不合法!");
	else if(n==2)
	{
		a[0][0]=1;
		a[0][1]=2;
		a[1][0]=2;
		a[1][1]=1;
	}
	else
	{
		sf(n/2);
		m=n/2;
		for(i=0;i<m;i++)
			for(j=0;j<m;j++)
			{
				a[i][j+m]=a[i][j]+m;
				a[i+m][j]=a[i][j]+m;
				a[i+m][j+m]=a[i][j];
			}	
		
	}
	return 0;
}



void main()
{

    printf("输入运动员个数:");
	scanf("%d",&n);
	s=sf(n);
	if(n%2==0)
	{
		for(i=0;i<n;i++)
		{
			for(j=0;j<n;j++)
			{
				printf("%4d",a[i][j]);
			}
			printf("\n");
		}
	}
	else
		printf("\n");

	getchar();
	getchar();
}
