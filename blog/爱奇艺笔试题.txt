#include "stdafx.h"
#include <iostream>

using namespace std;


int fun(int a,int b)
{
	static int m =1,i=2;
	i+= m +1;
	m = i +a +b;
	return m;
}
int _tmain(int argc, _TCHAR* argv[])
{

	int k = 5,m = 2,p;
	p = fun(k,m);
	cout<<p<<",";
	p = fun(k,m);
	cout<<p<<endl;
	return 0;
}
